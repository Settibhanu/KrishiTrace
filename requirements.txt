# ============================================================
# Krishi-Trace AI — Project Requirements
# ============================================================
# This is a Node.js / React project.
# Install dependencies using the commands below, NOT pip.
#
# Setup:
#   cd backend  && npm install
#   cd frontend && npm install
#
# Run:
#   cd backend  && npm run dev     (starts Express API on port 5000)
#   cd frontend && npm run dev     (starts Vite dev server on port 5173)
# ============================================================
